using namespace std;

    int score;

    cin >> score;

    char grade;

    grade = (score >= 90) ? 'A' :
            (score >= 80) ? 'B' :

#include <iostream>
using namespace std;

int main() {
    int score;
    char grade;

    cout << "Enter your score: ";
    cin >> score;

    // Step 1: Calculate grade using if-else
    if (score >= 90)
        grade = 'A';
    else if (score >= 80)
        grade = 'B';
    else if (score >= 70)
        grade = 'C';
    else if (score >= 60)
        grade = 'D';
    else
        grade = 'F';

    // Step 2: Print grade
    cout << "Your grade is " << grade << ". ";

    switch (grade) {
        case 'A':
            cout << "Excellent Work!";
            break;
        case 'B':
            cout << "Well done";
            break;
        case 'C':
            cout << "Good job";
            break;
        case 'D':
            cout << "You Passed, but you could do better";
            break;
        case 'F':
            cout << "Sorry, you failed";
            break;
        default:
            cout << "Invalid grade";
    }

    return 0;
}

#include <iostream>
using namespace std;

int main() {
    int score;
    char grade;

    cout << "Enter your score: ";
    cin >> score;

    // Step 1: Assign Grade
    if (score >= 90)
        grade = 'A';
    else if (score >= 80)
        grade = 'B';
    else if (score >= 70)
        grade = 'C';
    else if (score >= 60)
        grade = 'D';
    else
        grade = 'F';

    cout << "Your grade is " << grade << ". ";

    switch (grade) {
        case 'A':
            cout << "Excellent work! ";
            break;
        case 'B':
            cout << "Well done. ";
            break;
        case 'C':
            cout << "Good job. ";
            break;
        case 'D':
            cout << "You passed, but you could do better. ";
            break;
        case 'F':
            cout << "Sorry, you failed. ";
            break;
    }

    if (grade >= 'A' && grade <= 'D') {
        cout << "You are eligible for the next level.";
    } else {
        cout << "Please try again next time.";
    }

    return 0;
}
